1% @b_k:       rudder angle control signal 
 2% @psi_k:     measured compass angle 
 3% @init_data: initialization values
 4% returns:
 5% @b_est:     estimated rudder angle bias
 6% @psi_est:   estimated compass angle
 7% the init_data will be kept persistent while in loop
 8function [b_est, psi_est] = kalman_filter(b_in, psi_in, data)
 9%#codegen
10
11persistent init_flag x_pri P_pri Ad Bd Ed Cd Q R I 
12if isempty(init_flag)
13    init_flag = 1;
14    % distribute copies of init_data
15    [x_pri,P_pri,Ad,Bd,Ed,Cd,Q,R,I] = deal(data.x_pri0, data.P_pri0, ... 
16        data.Ad, data.Bd, data.Ed, data.Cd, data.Q, data.R, data.I)
17end
18
19% 1. compute Kalman gain:
20L = (P_pri*Cd')/((Cd*P_pri*Cd'+R));
21
22% 2. Update estimate with measurement (posteriori):
23x_post = x_pri + L*(psi_in-Cd*x_pri);
24
25% 3. Compute error variance for updated estimate:
26P = (I - L*Cd)*P_pri*(I-L*Cd)' + L*R*L';
27
28% 4. Project ahead (priori)
29x_pri = Ad*x_post + Bd*b_in;
30P_pri = Ad*P*Ad'+Ed*Q*Ed';
31
32% output
33psi_est = x_post(3); psi_est = x_post(5);
